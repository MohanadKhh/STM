16void GPIO_Init(GPIO_Struct *Port, GPIO_Config_t *conf)
17{
21 uint32_t Config_Data;

23 switch (conf->Mode)

25 case GPIO_Input_Analog:

28 case GPIO_Input_Float:
29 Config_Data = 0b0100;
30 break;
31 case GPIO_Input_Pullup:
32 Config_Data = 0b1000;
33 Port->GPIO_ODR |= conf->pin;
34 break;
35 case GPIO_Input_Pulldown:
36 Config_Data = 0b1000;
37 Port->GPIO_ODR &= ~(conf->pin);
38 break;
39 default:
40 Config_Data = conf->Speed | conf->Mode;
41 break;
42 }
43 // apply the configuration on the selected pins
44 for (uint16_t x = 0; x < 16; x++)
45 {
46 // if the pin is supposed to be configured
47 if (conf->pin & (1 << x))
48 {
49 // CRL is responsible fro pins 0-7
50 if (x < 8)
51 {
52 // clear the old configuration using the 4-bit mask
53 Port->GPIO_CRL &= ~(Config_Mask << (4 * x));
54 // apply the new configuration
55 Port->GPIO_CRL |= (Config_Data << (4 * x));
56 }
57 // CRH is responsible for pins 8-15
58 else
59 {
60 // clear the old configuration using the 4-bit mask
61 Port->GPIO_CRH &= ~(Config_Mask << (4 * (x - 8)));
62 // apply the new configuration
63 Port->GPIO_CRH |= (Config_Data << (4 * (x - 8)));
64 }
65 }
66 }
